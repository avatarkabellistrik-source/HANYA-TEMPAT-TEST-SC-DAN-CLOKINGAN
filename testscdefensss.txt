#!/bin/bash

TOKEN="8687462257:AAEzX81PQXglaSAcKi2pz1zg01Kb-XXBOQ4"
CHAT_ID="8259867883"

ROOT="/home/USERNAME/public_html"

NEW="/tmp/scan_new.txt"
OLD="/tmp/scan_old.txt"

# scan file mencurigakan
grep -rl "base64_decode\|system(\|exec(" $ROOT 2>/dev/null > $NEW

if [ -f "$OLD" ]; then

    NEWFILE=$(grep -Fxvf $OLD $NEW)
    DELFILE=$(grep -Fxvf $NEW $OLD)

    MSG=""

    if [ ! -z "$NEWFILE" ]; then
        MSG="🆕 FILE MENCURIGAKAN BARU:
$NEWFILE"
    fi

    if [ ! -z "$DELFILE" ]; then
        MSG="$MSG

🗑 FILE MENCURIGAKAN DIHAPUS:
$DELFILE"
    fi

    if [ ! -z "$MSG" ]; then
        curl -s -X POST "https://api.telegram.org/bot$TOKEN/sendMessage" \
        -d chat_id=$CHAT_ID \
        --data-urlencode text="$MSG"
    fi
fi

mv $NEW $OLD
