TOKEN="8687462257:AAEzX81PQXglaSAcKi2pz1zg01Kb-XXBOQ4"

ROOT="$HOME/public_html"

NEW="/tmp/scan_new.txt"
OLD="/tmp/scan_old.txt"

# scan file mencurigakan saja

# kalau sudah ada data lama

    # FILE BARU
    NEWFILE=$(comm -13 <(sort $OLD) <(sort $NEW))

    # FILE DIHAPUS
    DELFILE=$(comm -23 <(sort $OLD) <(sort $NEW))

    MSG=""

    if [ ! -z "$NEWFILE" ]; then
        MSG="$MSG\nđź†• FILE MENCURIGAKAN BARU:\n$NEWFILE"
    fi

    if [ ! -z "$DELFILE" ]; then
        MSG="$MSG\nđź—‘ FILE MENCURIGAKAN DIHAPUS:\n$DELFILE"
    fi

    if [ ! -z "$MSG" ]; then
        curl -s -X POST "https://api.telegram.org/bot$TOKEN/sendMessage" \
        -d chat_id=$CHAT_ID \
        -d text="$MSG"
    fi
fi

# simpan hasil scan sekarang
