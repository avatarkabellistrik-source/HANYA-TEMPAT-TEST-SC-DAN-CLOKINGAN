function sendTele($msg){

/* SCAN (DEEP) */
function scan($dir){
    global $watch_ext;

    $res = [];
    $files = @scandir($dir);
    if(!$files) return $res;

    foreach($files as $f){
        if($f=="."||$f=="..") continue;

        $path = $dir.'/'.$f;

        if(is_dir($path)){
            $res += scan($path);
        } else {
            $ext = strtolower(pathinfo($path, PATHINFO_EXTENSION));
            if(in_array($ext,$watch_ext)){
                $res[$path] = @filemtime($path);
            }
        }
    }
    return $res;
}

/* DETEKSI SHELL */
function detectShell($file){

    if(!file_exists($file)) return false;

    // skip file besar (biasanya legit)
    if(filesize($file) > 500000) return false;

    $c = @file_get_contents($file);
    if(!$c) return false;

    $score = 0;

    // 🚨 REMOTE LOADER
    if(preg_match('/https?:\/\/.+/i',$c)) $score+=5;
    if(preg_match('/file_get_contents\s*\(\s*[\'"]https?:\/\//i',$c)) $score+=10;

    // eval + external
    if(stripos($c,'eval(')!==false && preg_match('/https?:\/\//i',$c)) $score+=10;

    // command execution
    if(stripos($c,'system(')!==false) $score+=5;
    if(stripos($c,'exec(')!==false) $score+=5;
    if(stripos($c,'shell_exec')!==false) $score+=5;

    // obfuscation
    if(stripos($c,'base64_decode')!==false) $score+=4;
    if(stripos($c,'gzinflate')!==false) $score+=4;

    // kombinasi kuat
    if(preg_match('/base64_decode/i',$c) && preg_match('/eval/i',$c)) $score+=8;

    // input + exec
    if(preg_match('/\$_(GET|POST|REQUEST)/i',$c) &&
       preg_match('/system|exec|shell_exec/i',$c)){
        $score+=6;
    }

    // base64 panjang
    if(preg_match('/[A-Za-z0-9\/+]{200,}/',$c)){
        $score+=5;
    }

    return ($score >= 8) ? $score : false;
}
